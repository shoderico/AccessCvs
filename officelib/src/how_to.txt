2) ヘッダに __declspec(dllexport) or __declspec(dllimport) の定義を追加する

例）
    #include "officelib_global.h"
    // Define this symbol to __declspec(dllexport) or __declspec(dllimport)
    #ifndef VBIDE_EXPORT
    #define VBIDE_EXPORT OFFICELIB_SHARED_EXPORT
    #endif

2-2) inline を解除する

    *.h の inline を、XXXX_INLINE に置換する。
        ex) inline -> OFFICE_INLINE

    *.h の 先頭付近に、XXXX_INLINE の define を追加
        #if defined _MSC_VER
        #define OFFICE_INLINE    __declspec(noinline)
        #elif defined __MINGW32__
        #define OFFICE_INLINE
        #endif

    xxxx_impl.cpp を追加
    xxxx_impl.cpp に、#include を追加
        # 必要に応じて依存するヘッダーを include
        #include "office.h"
        
        namespace Office {

        }

    xxxx.h の inline の実装部分をすべて、xxxx_impl.cpp に移動
        xxxx.h
            // member function implementation     : ここから
            #ifndef QAX_DUMPCPP_OFFICE_NOINLINES
            ...
            #endif                                : ここまで

        xxxx_impl.cpp
        
        namespace Office {
            ここに貼り付け
        }


3) Mingwの場合：
    C++11 がサポートされない
    →　enum の forward declaration がサポートされない
    →　コンパイルエラーが出る
    ので、以下の対策をする

3-1) ヘッダーhの Q_DECLARE_OPAQUE_POINTER をコメントアウト
        access.h
        excel.h

3-2) ソースcppに #include <enum が定義されたヘッダ> を、自身のinclude 前に追加
        access.cpp
            #include "officelib/officelib.h"
            #include "access.h"
        excel.cpp
            #include "officelib/officelib.h"
            #include "excel.h"



4) Access::Application に以下のコンストラクタを追加
        access.cpp

    Application(_Application *iface)
    : QAxObject()

        initializeFrom(iface);
        delete iface;
    }
